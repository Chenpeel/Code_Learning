class KNN_C:
    def __init__(self,k) -> None:
        # 
        self.k = k
    
    def fit(self,X,y):
        '''
        
        
        ------
        X:  ,:[,]
            
        
        y:  ,:[]
            ,
        '''
        # Xndarray
        self.X = np.asarray(X) 
        self.y = np.asarray(y)
        
    def predict_basic(self,X):
        '''
         
        
        Parameters
        ------
        X:  ,:[,]
            
        
        Returns
        ------
        result: 
                
        '''
        X = np.asarray(X)
        result = []
        
        # 
        for x in X:
            # 
                # s = x - self.X    
            # 
            distance= np.sqrt(np.sum((x-self.X)**2,axis=1))
            # 
            index = distance.argsort()
            # K 
            index =index[:self.k]
            # y 
            count = np.bincount(self.y[index])
            # 
            result.append(count.argmax())
        return np.asarray(result)
    def predict(self,X):
        '''
         
        
        Parameters
        ------
        X:  ,:[,]
            
        
        Returns
        ------
        result: 
                
        '''
        X = np.asarray(X)
        result = []
        
        # 
        for x in X:
            # 
                # s = x - self.X    
            # 
            distance= np.sqrt(np.sum((x-self.X)**2,axis=1))
            # 
            index = distance.argsort()
            # K 
            index =index[:self.k]
            # y  
            count = np.bincount(self.y[index],weights=1/distance[index])
            # 
            result.append(count.argmax())
        return np.asarray(result)

class KNN_R:
    def __init__(self,k) -> None:
        # 
        self.k = k
    
    def fit(self,X,y):
        '''
        
        
        ------
        X:  ,:[,]
            
        
        y:  ,:[]
            ,
        '''
        # Xnparray
        self.X = np.asarray(X)
        self.y = np.asarray(y)
        
    def predict_basic(self,X):
        '''
         
        
        Parameters
        ------
        X:  ,:[,]
            
        
        Returns
        ------
        result: 
                
        '''
        X = np.asarray(X)
        result = []
        
        for x in X:
            # 
                # s = x - self.X    
            # 
            distance= np.sqrt(np.sum((x-self.X)**2,axis=1))
            # 
            index = distance.argsort()
            # K 
            index =index[:self.k]
            result.append(np.mean(self.y[index]))
        return result
    def predict(self,X):
        '''
         
        
        Parameters
        ------
        X:  ,:[,]
            
        
        Returns
        ------
        result: 
                
        '''
        X = np.asarray(X)
        result = []
        
        for x in X:
            # 
                # s = x - self.X    
            # 
            distance= np.sqrt(np.sum((x-self.X)**2,axis=1))
            # 
            index = distance.argsort()
            # K  
            index =index[:self.k]
            s = np.sum(1/(distance[index]+0.000000000000000000000001))
            weigths = (1/(distance[index]+0.000000000000000000000001)) /s
            
            result.append(np.sum(weigths *self.y[index]))
        return result
